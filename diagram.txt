Table Customers {
  customer_id int [pk, increment]
  first_name varchar(50)
  last_name varchar(50)
  email varchar(100) [unique]
  password varchar(255)
  phone_number varchar(20)
  address varchar(255)
  city varchar(100)
  state varchar(100)
  zip_code varchar(20)
}

Table Categories {
  category_id int [pk, increment]
  name varchar(100)
  description text
}

Table Products {
  product_id int [pk, increment]
  name string(100)
  description text
  price decimal(10, 2)
  stock_quantity int
  category_id int
  image_url varchar(255)
}

Table Orders {
  order_id int [pk, increment]
  customer_id int
  order_date datetime
  total_amount decimal(10, 2)
  shipping_address varchar(255)
  status varchar(50)
}

Table Order_Items {
  order_item_id int [pk, increment]
  order_id int
  product_id int
  quantity int
  price_at_purchase decimal(10, 2)
}

Table Payments {
  payment_id int [pk, increment]
  order_id int [unique]
  payment_method varchar(50)
  amount decimal(10, 2)
  payment_date datetime
  transaction_id varchar(100)
}

Table Shopping_Carts {
  cart_id int [pk, increment]
  customer_id int
  product_id int
  quantity int
  added_date datetime
}

// Relationships
Ref: Products.category_id > Categories.category_id
Ref: Orders.customer_id > Customers.customer_id
Ref: Order_Items.order_id > Orders.order_id
Ref: Order_Items.product_id > Products.product_id
Ref: Payments.order_id > Orders.order_id
Ref: Shopping_Carts.customer_id > Customers.customer_id
Ref: Shopping_Carts.product_id > Products.product_id


Ref: "Customers"."last_name" < "Customers"."email"
